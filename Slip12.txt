 using System;

class Program
{
    static void Main()
    {
        int n, first, last, sum;

        Console.Write("Enter Number: ");
        n = Convert.ToInt32(Console.ReadLine());

        last = n % 10;

        while (n >= 10)
        {
            n = n / 10;
        }

        first = n;

        sum = first + last;

        Console.WriteLine("Sum of First and Last Digit = " + sum);
    }
}

using System;
using System.Drawing;

namespace WebApplication1
{
    public partial class Slip12 : System.Web.UI.Page
    {
        protected void btnShow_Click(object sender, EventArgs e)
        {
            lblMessage.Text = txtName.Text + " : " + txtMessage.Text;

            lblMessage.Font.Bold = chkBold.Checked;
            lblMessage.Font.Italic = chkItalic.Checked;
            lblMessage.Font.Underline = chkUnderline.Checked;

            lblMessage.ForeColor = Color.Blue;
            lblMessage.BackColor = Color.LightYellow;
        }
    }
}
