using System;

class Program
{
    static void Main()
    {
        int[] arr = { 10, 50, 20, 80, 30 };
        int max = arr[0];

        foreach (int i in arr)
        {
            if (i > max)
                max = i;
        }

        Console.WriteLine("Largest Number = " + max);
    }
}
