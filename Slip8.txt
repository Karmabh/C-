Q1)C#
using System;

class Program
{
    static void Main()
    {
        int[] arr = { 10, 5, 80, 2, 50 };

        int min = arr[0];
        int max = arr[0];

        foreach (int i in arr)
        {
            if (i < min)
                min = i;

            if (i > max)
                max = i;
        }

        Console.WriteLine("Minimum = " + min);
        Console.WriteLine("Maximum = " + max);
    }
}
